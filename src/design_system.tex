% Local equivalent of needspace's non-starred command. It requests room for a
% block without boxing the content or disturbing the one-column reading order.
\newcommand{\Needspace}[1]{%
  \par
  \begingroup
    \dimen@=#1\relax
    \vskip\z@\@plus\dimen@
    \penalty-100
    \vskip\z@\@plus-\dimen@
    \vskip\dimen@
    \penalty9999
    \vskip-\dimen@
    \vskip\z@skip
  \endgroup
}
